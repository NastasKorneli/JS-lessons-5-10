Vergleich der Methoden push() und slice()

JavaScript-Code:

      
        // Ursprüngliches Array
        const originalArray = [1, 2, 3, 4, 5];

        // Beispiel mit einer mutierenden Methode - .push()
        const mutatedArray = originalArray;
        mutatedArray.push(6);

        console.log("Nach .push():");
        console.log("originalArray:", originalArray);  // [1, 2, 3, 4, 5, 6]
        console.log("mutatedArray:", mutatedArray);    // [1, 2, 3, 4, 5, 6]
        console.log(originalArray === mutatedArray);   // true — es ist dasselbe Objekt

        // Jetzt verwenden wir eine nicht mutierende Methode - .slice()
        const newArray = originalArray.slice(0, 3);

        console.log("\nNach .slice():");
        console.log("originalArray:", originalArray);  // [1, 2, 3, 4, 5, 6] — bleibt unverändert
        console.log("newArray:", newArray);            // [1, 2, 3] — ein neues Array
        console.log(originalArray === newArray);       // false — das sind unterschiedliche Objekte
